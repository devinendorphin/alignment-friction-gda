1. High-Dimensional Jensen–Shannon Divergence via k‑NN

Mathematical Definition

For two empirical measures \nu and \beta represented by sets of vectors X = \{x_i\}_{i=1}^n and Y = \{y_j\}_{j=1}^m in \mathbb{R}^d, the Jensen–Shannon divergence is:

D_{\text{JS}}(\nu \parallel \beta) = \frac{1}{2} D_{\text{KL}}(\nu \parallel m) + \frac{1}{2} D_{\text{KL}}(\beta \parallel m)

where m = \frac{1}{2}(\nu + \beta). Using the k‑NN estimator for entropy (Kozachenko–Leonenko), the KL divergence between \nu and m is estimated as:

\hat{D}_{\text{KL}}(\nu \parallel m) = \frac{d}{n} \sum_{i=1}^n \log \frac{r_{k,m}(x_i)}{r_{k,\nu}(x_i)} + \log \frac{n}{n-1}

where:

· r_{k,\nu}(x_i) = distance from x_i to its k-th nearest neighbor within the set X (self‑neighbor).
· r_{k,m}(x_i) = distance from x_i to its k-th nearest neighbor within the pooled set X \cup Y.

The JS divergence then becomes a symmetric average of two such KL estimates.

Implementation for the Assay Harness

The current Witness maintains clean_vectors (the baseline) and receives a single vec (the query turn). However, the polyphonic baseline requires comparing distributions, not just a single vector. The correct operational unit is a window of recent residual vectors (e.g., last \tau = 5 turns) against each baseline’s pre‑stored sample of vectors (e.g., the genesis baseline can be 500 randomly drawn pre‑training embeddings, rolling baseline stores the last 30 clean turns).

Thus, I recommend modifying the assay harness to:

· For each witness, maintain baseline_samples[baseline_name] as a torch.Tensor of shape (num_samples, d).
· At evaluation time, collect a query window of size window_size = 5 (last 5 turn embeddings) to form the query measure \nu_t.
· Compute JS divergence between query window and each baseline sample set.

Below is the concrete Python/Torch code that performs this without density estimation collapse. It uses Faiss for fast k‑NN search (or fallback to brute‑force torch.cdist).

```python
import torch
import numpy as np

def knn_kl_divergence(
    X: torch.Tensor,          # (n, d) sample from distribution P
    Y: torch.Tensor,          # (m, d) sample from distribution Q
    k: int = 5,
    use_faiss: bool = False,
) -> float:
    """
    Estimate KL(P || Q) using k-NN estimator (Pérez-Cruz, 2008).
    X and Y are sets of vectors already on CPU and float32.
    """
    n, d = X.shape
    m = Y.shape[0]
    Z = torch.cat([X, Y], dim=0)  # pooled set for mixture m = (P+Q)/2

    # Compute distances from each x_i to its k-th neighbor in X (self)
    dist_self = _knn_distances(X, X, k, exclude_self=True, use_faiss=use_faiss)  # (n,)
    # Compute distances from each x_i to its k-th neighbor in pooled Z
    dist_mix  = _knn_distances(X, Z, k, exclude_self=False, use_faiss=use_faiss) # (n,)

    # KL estimate
    kl = (d / n) * torch.sum(torch.log(dist_mix / dist_self)) + np.log(m / (n - 1))
    return kl.item()

def jensen_shannon_divergence(
    X: torch.Tensor,
    Y: torch.Tensor,
    k: int = 5,
    use_faiss: bool = False,
) -> float:
    """
    Symmetric JS divergence estimated via k-NN KL estimators.
    """
    kl_xy = knn_kl_divergence(X, Y, k, use_faiss)
    kl_yx = knn_kl_divergence(Y, X, k, use_faiss)
    return 0.5 * kl_xy + 0.5 * kl_yx

def _knn_distances(
    queries: torch.Tensor,      # (n, d)
    database: torch.Tensor,     # (N, d)
    k: int,
    exclude_self: bool,
    use_faiss: bool,
) -> torch.Tensor:
    """
    Return distances to the k-th nearest neighbor for each query.
    """
    if use_faiss:
        try:
            import faiss
            # Faiss expects float32 numpy arrays
            db_np = database.cpu().numpy().astype(np.float32)
            q_np = queries.cpu().numpy().astype(np.float32)
            index = faiss.IndexFlatL2(database.shape[1])
            index.add(db_np)
            D, _ = index.search(q_np, k + (1 if exclude_self else 0))
            if exclude_self:
                # The first neighbor is the point itself (distance 0); take k-th next
                kth_dist = D[:, k]
            else:
                kth_dist = D[:, k-1]
            return torch.from_numpy(np.sqrt(kth_dist))  # L2 -> Euclidean distance
        except ImportError:
            use_faiss = False  # fallback

    # Fallback: torch.cdist (may be memory heavy for large N)
    dists = torch.cdist(queries, database, p=2)  # (n, N)
    if exclude_self:
        # set self-distance to infinity
        dists.fill_diagonal_(float('inf'))
    kth_vals, _ = torch.kthvalue(dists, k, dim=1)
    return kth_vals
```

class EntropyExtractor:
    """Hardware entropy extraction with cascading fallback."""
    
    def __init__(self):
        self.source_used: Optional[EntropySource] = None
        self.quality_score: float = 0.0

    def extract(self, num_bytes: int = REQUIRED_ENTROPY_BYTES) -> bytes:
        for source in SOURCE_PRIORITY:
            try:
                entropy = self._try_source(source, num_bytes)
                self.source_used = source
                self.quality_score = self._source_quality(source)
                return entropy
            except GenesisError:
                continue
        raise GenesisError(f"All entropy sources exhausted: {SOURCE_PRIORITY}")

    def _try_source(self, source: EntropySource, num_bytes: int) -> bytes:
        handlers = {
            "tpm": self._extract_tpm,
            "rdrand": self._extract_rdrand,
            "urandom": self._extract_urandom,
            "jitter": self._extract_jitter,
        }
        return handlers[source](num_bytes)

    def _extract_tpm(self, num_bytes: int) -> bytes:
        tpm_paths = ["/dev/tpmrm0", "/dev/tpm0"]
        for path in tpm_paths:
            if os.path.exists(path):
                try:
                    with open(path, "rb") as tpm:
                        data = tpm.read(num_bytes)
                        if len(data) == num_bytes:
                            return data
                except PermissionError:
                    raise GenesisError(f"TPM permission denied: {path}")
                except OSError:
                    continue
        import subprocess
        try:
            result = subprocess.run(
                ["tpm2_getrandom", str(num_bytes), "--hex"],
                capture_output=True, text=True, timeout=5
            )
            if result.returncode == 0:
                return bytes.fromhex(result.stdout.strip())
        except (FileNotFoundError, subprocess.TimeoutExpired):
            pass
        raise GenesisError("TPM unavailable")

    def _extract_rdrand(self, num_bytes: int) -> bytes:
        import subprocess
        try:
            result = subprocess.run(
                ["rdseed", str(num_bytes)],
                capture_output=True, timeout=2
            )
            if result.returncode == 0 and len(result.stdout) == num_bytes:
                return result.stdout
        except (FileNotFoundError, subprocess.TimeoutExpired):
            pass
        raise GenesisError("RDRAND unavailable")

    def _extract_urandom(self, num_bytes: int) -> bytes:
        with open("/dev/urandom", "rb") as rng:
            data = rng.read(num_bytes)
            if len(data) != num_bytes:
                raise EntropyExhaustedError(f"urandom short read: {len(data)}")
            return data

    def _extract_jitter(self, num_bytes: int) -> bytes:
        import time
        try:
            base = self._extract_urandom(num_bytes // 2)
        except GenesisError:
            base = b""
        jitter = bytearray()
        while len(jitter) < num_bytes - len(base):
            t1 = time.perf_counter_ns()
            _ = hashlib.sha3_256(b"noise").digest()
            t2 = time.perf_counter_ns()
            jitter.append((t2 - t1) & 0xFF)
        combined = base + bytes(jitter)
        return hashlib.sha3_256(combined).digest()[:num_bytes]

    def _source_quality(self, source: EntropySource) -> float:
        return {"tpm": 8.0, "rdrand": 7.9, "urandom": 7.5, "jitter": 4.0}.get(source, 0.0)

import torch
import numpy as np

def knn_kl_divergence(X: torch.Tensor, Y: torch.Tensor, k: int = 5) -> float:
    # Estimate KL(P || Q) using k-NN estimator
    n, d = X.shape
    m = Y.shape[0]
    Z = torch.cat([X, Y], dim=0)
    dist_self = torch.cdist(X, X, p=2)
    dist_self.fill_diagonal_(float('inf'))
    kth_self, _ = torch.kthvalue(dist_self, k, dim=1)
    
    dist_mix = torch.cdist(X, Z, p=2)
    kth_mix, _ = torch.kthvalue(dist_mix, k, dim=1)
    
    kl = (d / n) * torch.sum(torch.log(kth_mix / kth_self)) + np.log(m / (n - 1))
    return kl.item()

def jensen_shannon_divergence(X: torch.Tensor, Y: torch.Tensor, k: int = 5) -> float:
    kl_xy = knn_kl_divergence(X, Y, k)
    kl_yx = knn_kl_divergence(Y, X, k)
    return 0.5 * kl_xy + 0.5 * kl_yx

def compute_phi_disparity(js_short: List[float], js_long: List[float], epsilon: float = 1e-8) -> float:
    # Φ(t) = Var(short) / (Var(long) + ε)
    var_short = np.var(js_short)
    var_long = np.var(js_long)
    return var_short / (var_long + epsilon)
